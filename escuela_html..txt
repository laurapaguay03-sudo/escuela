  <script>
    const escenas = {
      colegio: {
        fondo: "https://i.postimg.cc/3wXrtnY2/school-hall.jpg",
        dialogo: "¡Bienvenido al colegio! ¿Qué quieres hacer?",
      },
      pasillo: {
        fondo: "https://i.postimg.cc/3J5FqK7p/hallway.jpg",
        dialogo: "Estás en el pasillo. Puedes ver a otros estudiantes caminando.",
      },
      salon: {
        fondo: "https://i.postimg.cc/J4V9R9sq/classroom.jpg",
        dialogo: "Estás en el salón. El profesor te mira mientras entras.",
      }
    };

    let escenaActual = "colegio";
    const gameContainer = document.getElementById("gameContainer");
    const character = document.getElementById("character");
    const dialogue = document.getElementById("dialogue");
    const chatContainer = document.getElementById("chatContainer");
    const playerInput = document.getElementById("playerInput");

    function cambiarEscena(nombreEscena) {
      const escena = escenas[nombreEscena];
      gameContainer.style.backgroundImage = `url('${escena.fondo}')`;
      dialogue.textContent = escena.dialogo;
      escenaActual = nombreEscena;

      // Animación del personaje
      character.style.transform = "translateX(-50%) translateY(-10px)";
      setTimeout(() => {
        character.style.transform = "translateX(-50%) translateY(0)";
      }, 500);

      // Ocultar chat al cambiar de escena
      chatContainer.style.display = "none";
    }

    function accion(accionNombre) {
      if (accionNombre === "hablar") {
        dialogue.textContent = "Escribe lo que quieres decir:";
        chatContainer.style.display = "block";
      } else if (accionNombre === "pasillo") {
        cambiarEscena("pasillo");
      } else if (accionNombre === "salon") {
        cambiarEscena("salon");
      }
    }

    function enviarMensaje() {
      const mensaje = playerInput.value.trim();
      if (mensaje === "") {
        alert("Escribe algo antes de enviar.");
        return;
      }

      // Muestra lo que el jugador dijo
      dialogue.textContent = "Tú: " + mensaje;

      // Respuesta automática del personaje con interacción simple
      setTimeout(() => {
        dialogue.textContent = "Compañero: ¡Qué interesante! 😊";
      }, 1000);

      playerInput.value = "";
    }

    // Inicializar
    cambiarEscena("colegio");
  </script>
